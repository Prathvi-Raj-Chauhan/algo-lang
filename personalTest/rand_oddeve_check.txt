START
DECLARE num, loops, fact_num
loops = 0

OUTPUT "Welcome to the Number Facts Teller."

WHILE loops < 3
    OUTPUT "Please enter a number:"
    INPUT num
    fact_num = RANDOM(0,100)
    IF num > 0
        OUTPUT "That is a POSITIVE number."
        IF num > fact_num
            OUTPUT "That number is greater than ${fact_num}" 
        ENDIF
        ELSE
            OUTPUT "That number is less than ${fact_num}"
        ENDELSE
    ENDIF
    ELIF num < 0
        OUTPUT "That is a NEGATIVE number."
        OUTPUR "Fact number was ${fact_num}"
    ENDELIF
    ELSE
        OUTPUT "That is exactly ZERO."
        OUTPUR "Fact number was ${fact_num}"
    ENDELSE
    
    loops = loops + 1
ENDWHILE

OUTPUT "Analysis finished. Goodbye!"
END
